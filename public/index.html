🤖 TrashBot Plugin

Use .play song name in your WhatsApp bot:

const API = 'https://your-domain.up.railway.app';

export default async (m, { sock, text }) => {
  if (!text) return m.reply('Usage: .play <song>');
  await m.reply('🔍 Searching...');

  const res = await fetch(`${API}/downloader/youtubemp3?q=${encodeURIComponent(text)}&limit=1`);
  const [song] = await res.json();
  if (!song) return m.reply('❌ Not found.');

  await m.reply(`🎵 *${song.title}*\nDownloading...`);
  const dl = await fetch(song.download_url);
  const buffer = Buffer.from(await dl.arrayBuffer());

  await sock.sendMessage(m.chat, {
    audio: buffer, mimetype: 'audio/mpeg',
    fileName: song.title + '.mp3', ptt: false,
  }, { quoted: m });
};
